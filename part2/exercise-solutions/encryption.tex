\begin{questions}
    \item We claim that $x+2$ is its own inverse when $N = 2$ and $p = 4$, since
    \begin{align*}
        (x+2)^2 &= x^2 + 4x + 4\\
        &= x^0 + 4x + 4 & (\text{reduce powers modulo } N = 2)\\
        &= 4x + 5\\
        &= 1 & (\text{reduce coefficients modulo } p = 4)
    \end{align*}
\end{questions}
